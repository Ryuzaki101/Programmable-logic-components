********** Mapped Logic **********
FDCPE_Q: FDCPE port map (Q,'0','0',Q_CLR,state/state_SETF);
     Q_CLR <= (CLK AND R AND NOT S);
FDCPE_Qbar: FDCPE port map (Qbar,'1','0',state/state_SETF,Qbar_PRE);
     Qbar_PRE <= (CLK AND R AND NOT S);
state/state_SETF <= ((CLK AND R AND S)
      OR (CLK AND NOT R AND NOT S));

      FTCPE (Q,D,C,CLR,PRE,CE);
